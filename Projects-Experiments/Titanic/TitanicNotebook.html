def fill_in_age(cols):
    age = cols[0]
    pclass = cols[1]
    if pd.isnull(age):
        if pclass==1:
            return 38.23
        elif pclass==2:
            return 29.88
        else:
            return 25.14
        return age
    else:
        return age
