def fill_in_age(cols):
    age = cols[0]
    pclass = cols[1]
    if pd.isnull(age):
        if pclass==1:
            return 39.15
        elif pclass==2:
            return 29.50
        else:
            return 24.81
        return age
    else:
        return age
